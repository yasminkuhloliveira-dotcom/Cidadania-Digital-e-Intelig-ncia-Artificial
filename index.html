document.addEventListener('DOMContentLoaded', () => { // 1. Funcionalidade de Alternar Contraste (Modo Escuro) const toggleThemeBtn = document.getElementById('toggle-theme'); toggleThemeBtn.addEventListener('click', () => { document.body.classList.toggle('dark-mode'); // Atualiza a acessibilidade informando se o modo escuro está ativo const isDarkMode = document.body.classList.contains('dark-mode'); toggleThemeBtn.setAttribute('aria-label', isDarkMode ? "Alternar para modo claro" : "Alternar para modo escuro"); }); // 2. Funcionalidade de Validação do Quiz const quizForm = document.getElementById('quiz-form'); const quizFeedback = document.getElementById('quiz-feedback'); quizForm.addEventListener('submit', (event) => { // Evita que a página recarregue ao enviar o formulário event.preventDefault(); // Captura a opção selecionada pelo usuário const selectedOption = document.querySelector('input[name="quiz-answer"]:checked'); // Remove classes antigas de feedback para atualizar o estado quizFeedback.className = "feedback-box"; // Verifica se nenhuma opção foi marcada if (!selectedOption) { quizFeedback.textContent = "⚠️ Por favor, selecione uma alternativa antes de verificar!"; quizFeedback.classList.add('error'); return; } // Valida o valor da resposta informada if (selectedOption.value === 'deepfake') { quizFeedback.textContent = "🎉 Correto! A oscilação na voz e o cenário estático/artificial são características clássicas de uma deepfake."; quizFeedback.classList.add('success'); } else { quizFeedback.textContent = "❌ Incorreto. Um cenário estático e falhas artificiais na voz apontam para uma manipulação por IA (Deepfake)."; quizFeedback.classList.add('error'); } }); });
